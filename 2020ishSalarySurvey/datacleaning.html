Create Helper DF of Clean Column Names

We manually re-coded column names in the above google sheet, with one tab at the question level and one tab at the response level, to deal with multi-select questions that are stored spread across multiple columns.

Now, we need to join the question and response level tabs together to form a complete key, then use the values in that key to replace the column names currently on the main dataframe.

revert_names <- function(x){

  # check whether name needs to be fixed
  if(grepl("....", x, fixed = TRUE)){

    # remove the periods and number that read_sheet inserted
    vec <- stringr::str_remove(x, "\\.{4}\\d{2,}$")

    # add back the period that was in the original column name
    vec <- paste0(vec, ".")

  } else {
    vec <- x
  }

  return(vec)
}


# remove "....[number]" at the end of the column name
colnames(df) <- purrr::map_chr(
  colnames(df), revert_names
)

Recode Column Values

Alchemer stores the response to the multi-select question as the text of the response, but we want to transform these to a boolean for easy counting. For example, in the column negotiation_result.resp.increased_money, we set Yes, I increased my compensation to be TRUE. Missing values are NA.

# Among columns for multi-select responses, turn text to TRUE
# but ignore freetext columns labeld as "other_specify"

multi_cols <- colnames(df)[
  grepl(".resp.", colnames(df), fixed = TRUE) &
    ! grepl("other_specify", colnames(df), fixed = TRUE) &
    # Skills section should stay named as "Beginner, Advanced, etc" 
    ! grepl("skills_comfort", colnames(df), fixed = TRUE)
]

df <- df %>%
  mutate_at(
    all_of(multi_cols),
    ~ifelse(!is.na(.x), TRUE, NA)
  )

        # all columns have the same value 
        work_hist_duped == 1 | 
        # more years worked than is reasonable 
        age_minus_work < 18 | 
        # years worked is an extreme outlier
        years_worked_total %in% boxplot.stats(df$years_worked_total, coef = 2)$out | 
        years_worked_data %in% boxplot.stats(df$years_worked_data, coef = 3)$out | 
        years_worked_political_data %in% boxplot.stats(df$years_worked_political_data, coef = 3)$out | 
        years_worked_political %in% boxplot.stats(df$years_worked_political, coef = 3)$out | 
        # years worked in politics and in data listed as the same not-small number
        (identical(years_worked_political_data, years_worked_political) &
           years_worked_political_data > 4
        ) 
    )

recode_pd <- function(dat, pd_type){
  
  available <- paste0("prof_dev_available.resp.", {{pd_type}})
  used <- paste0("prof_dev_used.resp.", {{pd_type}})
  
  errs <- dat %>% filter(
    is.na(.data[[available]]), 
    .data[[used]] == TRUE
  ) %>% nrow()
  
  print(paste("There are", errs, "errors for", pd_type, ". Cleaning Now!"))

  dat <- dat %>% mutate(
    tmp = case_when(
      is.na(dat[[available]]) ~ NA,
      dat[[used]] == TRUE ~ TRUE,
      is.na(dat[[used]]) ~ FALSE
  ))

  return(dat$tmp)
}
